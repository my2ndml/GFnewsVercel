function loadRSS() { fetch("/api/get-rss") .then((response) => response.json()) .then((data) => { console.log("Dati ricevuti dall'API:", data); // Log dei dati if (Array.isArray(data) && data.length > 0) { const newsWall = document.getElementById('news-wall'); newsWall.innerHTML = ''; // Pulisce il contenitore dei feed // Crea i box delle notizie data.forEach((item) => { const newsItem = document.createElement('div'); newsItem.classList.add('news-item'); const title = document.createElement('h2'); title.textContent = item.title || "No title available"; newsItem.appendChild(title); const description = document.createElement('p'); description.textContent = item.description || "No Description available"; newsItem.appendChild(description); const link = document.createElement('a'); link.href = item.link || "#"; link.target = "_blank"; link.textContent = "Leggi di piÃ¹"; newsItem.appendChild(link); // Se esiste un'immagine, la aggiungiamo if (item.image) { const img = document.createElement('img'); img.src = item.image; img.alt = item.title || "Image"; img.style.width = "100%"; img.style.height = "auto"; newsItem.appendChild(img); } newsWall.appendChild(newsItem); }); } else { console.log("Dati non validi ricevuti:", data); document.getElementById('news-wall').innerHTML = "Nessuna notizia disponibile."; } }) .catch((error) => { console.error("Errore durante il caricamento del feed:", error); document.getElementById('news-wall').innerHTML = "Errore nel caricamento delle notizie."; }); }
